}

	var-bg-color: #fff;

}

Math

The Calc Function


:root {
	var-padding: 10px;
	var-margin: 20px;
}	
header {
	background: #fff;
	padding: calc(var(var-padding)*2);
	margin: calc((var(var-margin) + 20px)/2);
}

Mixing Units

	
header {
	background: #fff;
	width: calc(100% - 2em);

Supports


@supports (display:flex) {
	section {
		display: flex;
	}
}

Else Version


@supports not (display:flex) {
	section {
		display: block;
	}
}

Logical Operators


@supports (display:flex) or (-moz-display:flex) or (-webkit-display:flex){
	section {
		display: flex;
	}
}
@supports (transform: rotate(15deg)) and (filter: blur(5px)){
	aside img:hover {
		transform: rotate(15deg);
		filter: blur(5px);
	}
}
@supports ((-webkit-transform: rotate(15deg) 
	or -moz-transform: rotate(15deg) 
	or transform: rotate(15deg)) 
	and (filter: blur(5px){
	aside img:hover {
		-webkit-transform: rotate(15deg);
		-moz-transform: rotate(15deg);
		transform: rotate(15deg);
		filter: blur(5px);
	}
}

Animation


@keyframes swing {
    30% {
        -webkit-transform: translateX(-5px);
        transform: translateX(-5px);
    }
    50% {
        -webkit-transform: translateX(3px);
        transform: translateX(3px);
    }
    80% {
        -webkit-transform: translateX(2px);
        transform: translateX(2px);
    }
    100% {
        -webkit-transform: translateX(0);
        transform: translateX(0);
    }
}
.swing:hover
{
        animation: swing 1s ease;
        animation-iteration-count: 1;
}

Bonus

CSS Nesting


/* LESS Nesting */
section {
    li {
      color: #ccc;
      &:last-child { 
      	font-weight: 800; 
      }
    }
    p {
      color: #eee; 
      font-size: 0.875em
      line-height: 1.3em;
    }
  }

/* CSS Nesting */
section {{
    li {
      color: #ccc;
      {&:last-child { 
      	font-weight: 800; 
      }}
    }
    p {
      color: #eee; 
      font-size: 0.875em
      line-height: 1.3em;
    }
  }}

This is still just a work in progress, it's not supported by any browsers or anything.
